	\section{Question 1}\label{q1}
		For the following system of \gp{ls}
		\begin{subnumcases}{}
			\x{1}+2\x{2}+\x{3}=3\\
			-\x{1}+\x{2}+\x{3}=2\\
			\x{1}+\x{2}-x_3=1
		\end{subnumcases}

		\begin{enumerate}
			\item Write down the \gp{cofm}, the \g{convec} and the \g{augm}.
			\ans
				\begin{equation}
					A= 
					\begin{bmatrix}
					1 & 2 & 1 \\
					-1 & 1 & 2\\
					1 & 1 & -1
					\end{bmatrix}
				\end{equation}
				\begin{equation}
					B= 
					\begin{bmatrix}
						3\\
						2\\
						1
					\end{bmatrix}
				\end{equation}
				
				\begin{equation}
					[A|B]=
					\begin{amatrix}{3}
						1 & 2 & 1  & 3 \\
						-1 & 1 & 2 & 2\\
						1 & 1 & -1 & 1
					\end{amatrix}
				\end{equation}	
				
			\item Applying \gp{ro} on the \g{augm}, find the \g{solset}.
			\ans
			\begin{multline}
				[A|B]
				\rop{\Row{1}+\Row{2}\\-\Row{1}+\Row{3}}
				\begin{amatrix}{3}
					1 & 2 & 1 & 3\\
					0 & 3 & 3 & 5\\
					0 & -1 & -2 & -2
				\end{amatrix}
				\rop{\Row{2}\to\frac{1}{3}\Row{2}}
				\begin{bmatrix}{ccc|c}
					1 & 2 & 1 & 3\\
					0 & 1 & 1 & \frac{5}{3}\\
					0 & -1 & -2 & -\frac{5}{2}
				\end{bmatrix}
			\end{multline}
			
			
		\end{enumerate}
